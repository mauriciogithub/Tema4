# Los parmetros T, t_final y N son elegidos arbitrariamente

# bibliotecas para usar funciones estadisticas
import numpy as np
from scipy import stats
import matplotlib.pyplot as plt

# Variables aleatorias A y Z
vaC = stats.norm(5, np.sqrt(0.2))
vaW = stats.uniform(-2*np.pi*59.1, 2*np.pi*60.1)
vaZ = stats.uniform(0, np.pi/2)

# Creacin del vector de tiempo
T = 100			# nmero de elementos
t_final = 10	# tiempo en segundos
t = np.linspace(0, t_final, T)

# Inicializacin del proceso aleatorio X(t) con N realizaciones
N = 10
X_t = np.empty((N, len(t)))	# N funciones del tiempo x(t) con T puntos

# Creacin de las muestras del proceso x(t) (A y Z independientes)
for i in range(N):
	C = vaC.rvs()    
	Z = vaZ.rvs()
	x_t = C * np.cos(2*59.6*np.pi*t + Z)
	X_t[i,:] = x_t
	plt.plot(t, x_t)

# Promedio de las N realizaciones en cada instante (cada punto en t)
P = [np.mean(X_t[:,i]) for i in range(len(t))]
plt.plot(t, P, lw=6)

# Graficar el resultado terico del valor esperado
E = 6/np.pi * np.cos(np.pi*t)
plt.plot(t, E, '-.', lw=4)

# Mostrar las realizaciones, y su promedio calculado y terico
plt.title('Realizaciones del proceso aleatorio $X(t)$')
plt.xlabel('$t$')
plt.ylabel('$x_i(t)$')
plt.show()

# T valores de desplazamiento tau
desplazamiento = np.arange(T)
taus = desplazamiento/t_final

# Inicializacin de matriz de valores de correlacin para las N funciones
corr = np.empty((N, len(desplazamiento)))

# Nueva figura para la autocorrelacin
plt.figure()

# Clculo de correlacin para cada valor de tau
for n in range(N):
	for i, tau in enumerate(desplazamiento):
		corr[n, i] = np.correlate(X_t[n,:], np.roll(X_t[n,:], tau))/T
	plt.plot(taus, corr[n,:])

# Valor terico de correlacin
Rxx = 19/2 * np.cos(np.pi*taus)

# Grficas de correlacin para cada realizacin y la
plt.plot(taus, Rxx, '-.', lw=4, label='Correlacin terica')
plt.title('Funciones de autocorrelacin de las realizaciones del proceso')
plt.xlabel(r'$\tau$')
plt.ylabel(r'$R_{XX}(\tau)$')
plt.legend()
plt.show()
